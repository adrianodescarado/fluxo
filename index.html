⚙️ Setup Supabase

Para desativar confirmação de email:
Supabase → Authentication → Providers → Email → desative "Confirm email"

Cole este SQL no Supabase → SQL Editor e clique Run:

-- 1. TABELA DE FLUXOS
create table if not exists mandrake_flows (
  id text primary key,
  user_id text not null,
  name text not null,
  channel text default 'wpp',
  status text default 'draft',
  description text default '',
  nodes_json text default '[]',
  edges_json text default '[]',
  created_at timestamptz default now(),
  updated_at timestamptz default now()
);
alter table mandrake_flows enable row level security;
drop policy if exists "Users manage own flows" on mandrake_flows;
drop policy if exists "Public can read flows" on mandrake_flows;
create policy "Users manage own flows"
  on mandrake_flows for all
  using (auth.uid()::text = user_id);
create policy "Public can read flows"
  on mandrake_flows for select using (true);

-- 2. TABELA DE ANALYTICS
create table if not exists mandrake_analytics (
  id bigserial primary key,
  flow_id text, user_id text, event text,
  created_at timestamptz default now()
);
alter table mandrake_analytics enable row level security;
drop policy if exists "Insert analytics" on mandrake_analytics;
drop policy if exists "Read analytics" on mandrake_analytics;
create policy "Insert analytics"
  on mandrake_analytics for insert with check (true);
create policy "Read analytics"
  on mandrake_analytics for select using (true);

-- 3. BUCKET DE AVATARES
insert into storage.buckets (id, name, public)
  values ('mandrake-avatars','mandrake-avatars',true)
  on conflict do nothing;
